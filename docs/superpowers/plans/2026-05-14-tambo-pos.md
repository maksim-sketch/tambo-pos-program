# Tambo POS Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a demo-ready, multi-tenant capable POS app with a Tambo-driven conversational UI that renders cart, inventory, customer, and reporting components from natural language commands.

**Architecture:** Use a Bun workspace monorepo with `apps/api`, `apps/web`, and `packages/shared`. Execute the work in a Tambo-first order: bootstrap the workspace, ship the React retail shell plus Tambo registry early, prove the cashier flow in mock mode, then harden the Hono API, tenant-scoped persistence, checkout mutations, and SSE inventory sync behind it.

**Tech Stack:** Bun, TypeScript strict mode, Hono, Drizzle, SQLite, React, Vite, Tailwind CSS, Zod, `@tambo-ai/react`, Vitest, Testing Library, Playwright.

```ts
import { and, eq, like } from "drizzle-orm";
import { InventoryStatusResponseSchema } from "@shared/contracts/inventory";
import { branches, inventoryLevels, products, tenants } from "../db/schema";

export async function getInventorySnapshot(db: any, input: {
  tenantSlug: string;
  productQuery: string;
}) {
  const rows = await db
    .select({
      productId: products.id,
      productName: products.name,
      branchId: branches.id,
      branchCode: branches.code,
      branchName: branches.name,
      quantity: inventoryLevels.quantity,
    })
    .from(inventoryLevels)
    .innerJoin(products, eq(products.id, inventoryLevels.productId))
    .innerJoin(branches, eq(branches.id, inventoryLevels.branchId))
    .innerJoin(tenants, eq(tenants.id, inventoryLevels.tenantId))
    .where(
      and(
        eq(tenants.slug, input.tenantSlug),
        like(products.name, `%${input.productQuery}%`),
      ),
    );

  const [first] = rows;
  return InventoryStatusResponseSchema.parse({
    productId: first?.productId ?? "",
    productName: first?.productName ?? input.productQuery,
    branches: rows.map((row) => ({
      branchId: row.branchId,
      branchCode: row.branchCode,
      branchName: row.branchName,
      quantity: row.quantity,
      lowStock: row.quantity <= 3,
    })),
    lastUpdatedAt: new Date().toISOString(),
  });
}
```

```ts
type InventoryEvent = {
  tenantSlug: string;
  branchCode: string;
  productName: string;
  quantityDelta: number;
};

const listeners = new Set<(event: InventoryEvent) => void>();

export const inventoryBus = {
  publish(event: InventoryEvent) {
    listeners.forEach((listener) => listener(event));
  },
  subscribe(listener: (event: InventoryEvent) => void) {
    listeners.add(listener);
    return () => listeners.delete(listener);
  },
};
```

Create `apps/api/src/services/sales-service.ts`:

```ts
import { eq } from "drizzle-orm";
import { inventoryEvents, inventoryLevels, saleItems, sales } from "../db/schema";
import { inventoryBus } from "../realtime/inventory-bus";

export async function checkoutSale(db: any, input: {
  tenantSlug: string;
  branchCode: string;
  customerId?: string;
  items: Array<{ productName: string; quantity: number }>;
}) {
  const tenant = await db.query.tenants.findFirst({
    where: (table: any, { eq: equal }: any) => equal(table.slug, input.tenantSlug),
  });

  if (!tenant) {
    throw new Error("Invalid tenant");
  }

  const branch = await db.query.branches.findFirst({
    where: (table: any, { and: both, eq: equal }: any) =>
      both(equal(table.code, input.branchCode), equal(table.tenantId, tenant.id)),
  });

  if (!branch) {
    throw new Error("Invalid branch");
  }

  const saleId = crypto.randomUUID();
  let totalCents = 0;
  const persistedItems: Array<{ productId: string; quantity: number; unitPriceCents: number }> = [];

  for (const item of input.items) {
    const product = await db.query.products.findFirst({
      where: (table: any, { and: both, eq: equal }: any) =>
        both(equal(table.name, item.productName), equal(table.tenantId, tenant.id)),
    });

    if (!product) {
      throw new Error(`Unknown product: ${item.productName}`);
    }

    totalCents += product.priceCents * item.quantity;
    persistedItems.push({
      productId: product.id,
      quantity: item.quantity,
      unitPriceCents: product.priceCents,
    });

    const current = await db.query.inventoryLevels.findFirst({
      where: (table: any, { and: both, eq: equal }: any) =>
        both(equal(table.productId, product.id), equal(table.branchId, branch.id)),
    });

    if (!current || current.quantity < item.quantity) {
      throw new Error(`Insufficient inventory for ${item.productName}`);
    }

    await db
      .update(inventoryLevels)
      .set({ quantity: current.quantity - item.quantity })
      .where(eq(inventoryLevels.id, current.id));

    await db.insert(inventoryEvents).values({
      id: crypto.randomUUID(),
      tenantId: tenant.id,
      branchId: branch.id,
      productId: product.id,
      quantityDelta: -item.quantity,
      reason: "sale",
      createdAt: new Date().toISOString(),
    });

    inventoryBus.publish({
      tenantSlug: input.tenantSlug,
      branchCode: input.branchCode,
      productName: item.productName,
      quantityDelta: -item.quantity,
    });
  }

  await db.insert(sales).values({
    id: saleId,
    tenantId: tenant.id,
    branchId: branch.id,
    customerId: input.customerId ?? null,
    totalCents,
    createdAt: new Date().toISOString(),
  });

  await db.insert(saleItems).values(
    persistedItems.map((item) => ({
      id: crypto.randomUUID(),
      saleId,
      productId: item.productId,
      quantity: item.quantity,
      unitPriceCents: item.unitPriceCents,
    })),
  );

  return {
    saleId,
    totalItems: input.items.reduce((sum, item) => sum + item.quantity, 0),
    totalCents,
    message: "Sale completed",
  };
}
```

```ts
import { eq } from "drizzle-orm";
import { branches, products, saleItems, sales } from "../db/schema";

export async function getCustomerProfile(db: any, input: {
  tenantSlug: string;
  search: string;
}) {
  const tenant = await db.query.tenants.findFirst({
    where: (table: any, { eq: equal }: any) => equal(table.slug, input.tenantSlug),
  });

  if (!tenant) {
    throw new Error("Invalid tenant");
  }

  const customer = await db.query.customers.findFirst({
    where: (table: any, { and: both, eq: equal, like }: any) =>
      both(equal(table.tenantId, tenant.id), like(table.fullName, `%${input.search}%`)),
  });

  if (!customer) {
    throw new Error(`Customer not found: ${input.search}`);
  }

  const purchases = await db
    .select({
      branchName: branches.name,
      itemName: products.name,
      totalCents: saleItems.quantity,
      unitPriceCents: saleItems.unitPriceCents,
    })
    .from(sales)
    .innerJoin(branches, eq(branches.id, sales.branchId))
    .innerJoin(saleItems, eq(saleItems.saleId, sales.id))
    .innerJoin(products, eq(products.id, saleItems.productId))
    .where(eq(sales.customerId, customer.id));

  return {
    customer,
    branchesVisited: [...new Set(purchases.map((purchase) => purchase.branchName))],
    recentPurchases: purchases.map((purchase) => ({
      branchName: purchase.branchName,
      itemName: purchase.itemName,
      totalCents: purchase.totalCents * purchase.unitPriceCents,
    })),
    totalSpendCents: purchases.reduce(
      (sum, purchase) => sum + purchase.totalCents * purchase.unitPriceCents,
      0,
    ),
  };
}
```

```ts
import { eq, sql } from "drizzle-orm";
import { customers, products, saleItems, sales } from "../db/schema";

export async function getDailySalesReport(db: any, input: { tenantSlug: string }) {
  const tenant = await db.query.tenants.findFirst({
    where: (table: any, { eq: equal }: any) => equal(table.slug, input.tenantSlug),
  });

  if (!tenant) {
    throw new Error("Invalid tenant");
  }

  const series = await db
    .select({
      hour: sql<string>`strftime('%H:00', ${sales.createdAt})`,
      totalCents: sql<number>`sum(${sales.totalCents})`,
    })
    .from(sales)
    .where(eq(sales.tenantId, tenant.id))
    .groupBy(sql`strftime('%H:00', ${sales.createdAt})`);

  const topItems = await db
    .select({
      productName: products.name,
      quantity: sql<number>`sum(${saleItems.quantity})`,
      revenueCents: sql<number>`sum(${saleItems.quantity} * ${saleItems.unitPriceCents})`,
    })
    .from(saleItems)
    .innerJoin(products, eq(products.id, saleItems.productId))
    .innerJoin(sales, eq(sales.id, saleItems.saleId))
    .where(eq(sales.tenantId, tenant.id))
    .groupBy(products.name);

  const highValueCustomers = await db
    .select({
      fullName: customers.fullName,
      totalSpendCents: sql<number>`sum(${sales.totalCents})`,
    })
    .from(sales)
    .innerJoin(customers, eq(customers.id, sales.customerId))
    .where(eq(sales.tenantId, tenant.id))
    .groupBy(customers.fullName);

  return {
    tenantSlug: input.tenantSlug,
    range: "today",
    series,
    topItems,
    highValueCustomers,
  };
}
```

```ts
const API_BASE = import.meta.env.VITE_API_BASE_URL;

export async function apiGet<T>(path: string): Promise<T> {
  const response = await fetch(`${API_BASE}${path}`);
  if (!response.ok) throw new Error(`GET ${path} failed`);
  return response.json();
}

export async function apiPost<T>(path: string, body: unknown): Promise<T> {
  const response = await fetch(`${API_BASE}${path}`, {
    method: "POST",
    headers: { "content-type": "application/json" },
    body: JSON.stringify(body),
  });
  if (!response.ok) throw new Error(`POST ${path} failed`);
  return response.json();
}
```
